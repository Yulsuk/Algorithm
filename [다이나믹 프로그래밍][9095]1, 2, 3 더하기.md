#include <stdio.h>

int d[11];

int dp(int a)
{
	d[1] = 1;
	d[2] = 2;
	d[3] = 4;
	
	int i;
	for(i=4; i<=a; i++)
	{
		d[i] = d[i-1] + d[i-2] + d[i-3];
	}
	
	return d[a];
}

int main()
{
	int T;
	scanf("%d", &T);
	
	int i;
	int n;
	for(i=0; i<T; i++)
	{
		scanf("%d", &n);
		printf("%d\n", dp(n));
	}
	
	return 0;
}
